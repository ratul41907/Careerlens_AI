# ============================================================================
# Installation Instructions
# ============================================================================
#
# 1. Create virtual environment:
#    python -m venv venv
#    venv\Scripts\activate (Windows)
#    source venv/bin/activate (Mac/Linux)
#
# 2. Install dependencies:
#    pip install -r requirements.txt
#
# 3. Download ML models:
#    python -m spacy download en_core_web_sm
#    python -c "from sentence_transformers import SentenceTransformer; SentenceTransformer('all-MiniLM-L6-v2')"
#
# 4. Install Tesseract OCR (for document extraction):
#    Windows: Download from https://github.com/UB-Mannheim/tesseract/wiki
#    Mac: brew install tesseract
#    Linux: sudo apt-get install tesseract-ocr
#
# 5. (Optional) Install Ollama for LLM features:
#    Download from https://ollama.ai/
#    Run: ollama pull gemma2:2b
#
# ============================================================================

#

#

#
# Issue: docx2pdf not working
# Solution: Requires MS Word (Windows) or use LibreOffice fallback
#
